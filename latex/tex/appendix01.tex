\chapter{Appendix}

\section*{Resources review}
\subsection*{Apps}
    \begin{itemize}
        \item \cite{Duolingo}: Wasn´t very useful to me, felt like I stopped learning and started just going through the motions.
        \item \cite{Memrise}: I liked the short video clips and the AI tutor.        
    \end{itemize}  

\subsection*{Books}
    \begin{itemize}
        \item \cite{Ragusa}: Very old-fashioned, but the book is slim and quick to read.
    \end{itemize}  

\subsection*{Online}
    \begin{itemize}
        \item \cite{Cangiano}: Blog site with friendly explanations, e.g. of \iti{grande}.
        \item \cite{Capilli}: I found \iti{molto} confusing, but this site explained it well.
        \item \cite{Polyglottist}: Explanation of adjectives and their placement. Bit dry.
        \item \cite{Quesito}: Comparing \iti{essere} and \iti{stare} in a simple way.
        \item \cite{Wordreference}: I use this site all the time to look things up, super useful multi-lingual dictionary and verb tool.        
    \end{itemize}
